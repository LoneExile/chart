Thank you for installing {{ .Chart.Name }}.

Your release is named {{ .Release.Name }}.

To get the application URL, run these commands:

{{- if contains "NodePort" .Values.gotrue.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Release.Name }}-gotrue)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.gotrue.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ .Release.Name }}-gotrue'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Release.Name }}-gotrue --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.gotrue.service.port }}
{{- else if contains "ClusterIP" .Values.gotrue.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app=gotrue" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:9999 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9999:9999
{{- end }}

AI Service is available at:
{{- if contains "ClusterIP" .Values.ai.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ .Release.Name }}-ai {{ .Values.ai.service.port }}:{{ .Values.ai.service.port }}
{{- end }}
